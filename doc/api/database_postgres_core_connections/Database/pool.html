Future<Pool<Connection>> get pool async {
  if (!_inilized) {
    throw StateError(
      'database is not initilized: did you forget to call Database.init',
    );
  }
  return _pool ??= Pool<Connection>.withEndpoints(
    _endpoints!,
    settings: _poolSetting,
  );
}
